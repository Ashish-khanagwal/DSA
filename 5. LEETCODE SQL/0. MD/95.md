# Code

```mysql []
with weekly AS (
    SELECT employee_id,
        DATE_SUB(meeting_date, INTERVAL WEEKDAY(meeting_date) DAY) AS week_start,
        SUM(duration_hours) AS total_hours
    FROM meetings
    GROUP BY employee_id, week_start
)
SELECT e.employee_id,
    e.employee_name,
    e.department,
    SUM(
        CASE WHEN w.total_hours > 20 THEN 1 ELSE 0 END
    ) AS meeting_heavy_weeks
FROM employees e
JOIN weekly w
    ON e.employee_id = w.employee_id
GROUP BY e.employee_id
HAVING meeting_heavy_weeks >= 2
ORDER BY meeting_heavy_weeks DESC, e.employee_name ASC;
```
